#!/bin/bash
change_filename() {
for d in *; do
  # echo $d
  if [ -d "$d" ]; then
    (cd -- "$d" && change_filename)
  fi
  newdate=$(stat -f %SB -t %Y__%m__%d "$d")
  filepath=$(realpath "$d")
  arrIN=(${filepath//\// })
  filename=${newdate}____${arrIN[1]}
  newname=`echo $filename | sed -e 's/ /_/g'`;
  # echo $newname
  mv "$d" "$newname"
done
}
