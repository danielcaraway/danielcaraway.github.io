import matplotlib.pyplot as plt
import matplotlib
import statistics
matplotlib.rcParams['pdf.fonttype'] = 42
matplotlib.rcParams['ps.fonttype'] = 42

colors = np.array(['red', 'orange', 'yellow', 'green', 'blue', 'purple', 'pink', 'brown', 'black', 'salmon'])
def make_scatterplot(for_scatterplot):
    studio, studio_name = for_scatterplot
    studio = studio.sort_values(by='budget',ascending=False)
    studio_movies = [attr for attr in studio['title']]
    studio_budget = [attr for attr in studio['budget']]
#     avg_budget = sum(studio_budget)/len(studio_budget)
#     avg_budget = statistics.median(studio_budget)
    avg_budget = np.percentile(studio_budget, 85) 
    studio_percent_profit = [attr for attr in studio['dom_percent_profit']]
#     avg_pp = sum(studio_percent_profit)/len(studio_percent_profit)
#     avg_pp = statistics.median(studio_percent_profit)
    avg_pp = np.percentile(studio_percent_profit, 85)
    studio_genre = [attr for attr in studio['genre']]
    unique_genres = np.unique(studio_genre, return_inverse=True)[1].tolist()
    genres = np.array(unique_genres)

    movies = studio_movies
    x_coords = studio_budget
    y_coords = studio_percent_profit

    f = plt.figure()
    for i,movie in enumerate(movies):
        x = x_coords[i]
        y = y_coords[i]
        plt.scatter(x, y, marker='o', c=colors[genres][i], label=studio_genre[i] if studio_genre[i] not in plt.gca().get_legend_handles_labels()[1] else '')
        if len(movies) < 4:
            plt.text(x+0.5, y+0.5, movie, fontsize=9)
        else:
            if x > avg_budget:
                plt.text(x+0.5, y+0.5, movie, fontsize=9)
            if (y > avg_pp) and (x < avg_budget):
                plt.text(x+0.5, y+0.5, movie, fontsize=9)

    plt.xlabel('Budget (in millions)')
    plt.ylabel('Percent Profit')
    plt.legend(loc = "lower center", ncol=2, title=studio_name + " " + str(year) + " Portfolio", bbox_to_anchor = (.5, -.5))
    plt.show()
    p = plt.show()
    f.savefig(studio_name + "_" + str(year) + ".pdf", bbox_inches='tight')
